#!/usr/bin/env python3 """ split_chat.py — Split chat.min.json into separate JSON files in current folder Usage: python3 split_chat.py [input.json] [year|size] [messages_per_file] """ import json import os import sys from collections import defaultdict import re def split_by_year(input_file="chat.min.json", output_dir="."): """Split messages into separate JSON files by year in current folder""" print(f"Reading {input_file}...") with open(input_file, 'r', encoding='utf-8') as f: data = json.load(f) fields = data['fields'] rows = data['rows'] # Group messages by year year_messages = defaultdict(list) timestamp_idx = fields.index('timestamp') for row in rows: timestamp = row[timestamp_idx] match = re.match(r'\d{2}\.(\d{2})\.(\d{4})', timestamp) if match: year = match.group(2) year_messages[year].append(row) else: year_messages['unknown'].append(row) # Create index file in current directory index = { "v": 1, "type": "index", "files": [], "total_messages": len(rows) } # Save each year as separate file in current directory for year in sorted(year_messages.keys()): if year == 'unknown': filename = "chat_unknown.json" display_name = "Unknown Date" else: filename = f"chat_{year}.json" display_name = year output_data = { "v": 1, "fields": fields, "rows": year_messages[year], "metadata": { "year": year if year != 'unknown' else None, "message_count": len(year_messages[year]), "display_name": display_name } } with open(filename, 'w', encoding='utf-8') as f: json.dump(output_data, f, ensure_ascii=False, separators=(',', ':')) print(f" ✓ Saved {len(year_messages[year]):,} messages to {filename}") index["files"].append({ "name": display_name, "filename": filename, "message_count": len(year_messages[year]), "year": year if year != 'unknown' else None }) # Save index file in current directory with open("index.json", 'w', encoding='utf-8') as f: json.dump(index, f, ensure_ascii=False, separators=(',', ':')) print(f"\n✅ Split complete!") print(f" Total messages: {len(rows):,}") print(f" Files created: {len(year_messages)}") print(f" Index file: index.json") print(f"\n📁 All files are in the current directory (same as your HTML)") return index def split_by_custom_size(input_file="chat.min.json", messages_per_file=5000): """Split by message count in current folder""" print(f"Reading {input_file}...") with open(input_file, 'r', encoding='utf-8') as f: data = json.load(f) fields = data['fields'] rows = data['rows'] total = len(rows) num_files = (total + messages_per_file - 1) // messages_per_file index = { "v": 1, "type": "index", "files": [], "total_messages": total } for i in range(num_files): start = i * messages_per_file end = min((i + 1) * messages_per_file, total) filename = f"chat_part_{i+1:03d}.json" display_name = f"Messages {start+1:,} - {end:,}" output_data = { "v": 1, "fields": fields, "rows": rows[start:end], "metadata": { "part": i + 1, "start_index": start, "end_index": end, "message_count": end - start, "display_name": display_name } } with open(filename, 'w', encoding='utf-8') as f: json.dump(output_data, f, ensure_ascii=False, separators=(',', ':')) print(f" ✓ Saved {end-start:,} messages to {filename}") index["files"].append({ "name": display_name, "filename": filename, "message_count": end - start, "part": i + 1 }) with open("index.json", 'w', encoding='utf-8') as f: json.dump(index, f, ensure_ascii=False, separators=(',', ':')) print(f"\n✅ Split complete! Created {num_files} files in current directory.") return index if __name__ == "__main__": input_file = sys.argv[1] if len(sys.argv) > 1 else "chat.min.json" split_type = sys.argv[2] if len(sys.argv) > 2 else "year" if split_type == "year": split_by_year(input_file) elif split_type == "size": size = int(sys.argv[3]) if len(sys.argv) > 3 else 5000 split_by_custom_size(input_file, size) else: print("Usage: python3 split_chat.py [input.json] [year|size] [messages_per_file]")
